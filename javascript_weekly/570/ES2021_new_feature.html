    # 逻辑赋值操作符(&&= ||= ??=)

    逻辑赋值操作符📖 (opens new window). 

    // 或 或等
x ||= y;
x || (x = y);

// 与 与等
x &&= y;
x && (x = y);

// 零合并
x ??= y;
x ?? (x = y);

    # Promise.any 和错误合集

    Promise.any + 错误合集📖 (opens new window).

    Promise.any([
  fetch('https://v8.dev/').then(() => 'home'),
  fetch('https://v8.dev/blog').then(() => 'blog'),
  fetch('https://v8.dev/docs').then(() => 'docs')
]).then((first) => {
 	// 任意一个 promise 状态为 fulfilled
  console.log(first);
  // → 'home'
}).catch((error) => {
  // 所有 promise 状态都为 rejected
  console.log(error);
});
